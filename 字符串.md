##### 字符串反序 KY5

```C++
#include<iostream>
using namespace std;

int main(){
    
    int x, y;
    while(cin >> x >> y){
        while(x!=y){
            if(x < y) y/=2;
            else x/=2;
        }
        cout << x << endl;
    }
}
```

##### 单词替换 KY46

```C++
#include<iostream>
#include<string>
#include<vector>

using namespace std;
vector<string> v;

int main(){
    string s;
    while(cin >> s){
        v.push_back(s);
        if(getchar() == '\n') break;  //匹配缓冲区中的回车
    }
    string s1, s2;
    cin >> s1 >> s2;
    for(int i = 0; i < v.size(); ++i){
        if(v[i] == s1) cout << s2 << " ";
        else cout << v[i] << " ";
    }
    cout << endl;
}
```

##### 密码翻译 KY33

```C++
#include<iostream>
#include<string>
using namespace std;

string code(string s){
    for(int i = 0; i < s.size(); ++i){
        if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')){
            s[i]++;
            //溢出处理  小写字母的ASCII码在大写字母的后面
            if(s[i] > 'Z' && s[i] < 'a' || s[i] > 'z'){
                s[i] -= 26;
            }
        }
    }
    return s;
}

int main(){
    string s;
    getline(cin, s);
    string ans = code(s);
    cout << ans << endl;
}
```

##### 首字母大写 KY57

```C++
// 先处理首个单词  再遍历剩余的字符串
#include<iostream>
#include<string>

using namespace std;

int main(){
    string s;
    
    while(getline(cin, s)){
        //处理首字母
        if(s[0] >= 'a' && s[0] <= 'z') s[0] = s[0] - 32;
        
        for(int i = 1; i < s.size(); ++i){
            if((s[i-1] == ' ' || s[i-1] == '\t' ||
              s[i-1] == '\r' || s[i-1] == '\n') && (s[i] >= 'a' && s[i] <= 'z')){
                s[i] -= 32;
            }
              
        }
        cout << s << endl;
    }
} 
```

##### 子串计算 KY68

```C++
#include<iostream>
#include<map>
#include<string>
using namespace std;

int main(){
    map<string, int> m;
    string str;
    cin >> str;
    for(int i = 0; i < str.size(); ++i){
        for(int j = i; j < str.size(); ++j){
            m[str.substr(i,j-i+1)]++;
        }
    }
    for(map<string, int>::iterator it = m.begin(); it != m.end(); ++it){
        if(it->second > 1) cout << it->first << " " << it->second << endl;
    }
    return 0;
}
```
