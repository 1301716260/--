##### 字符串反序 KY5

```C++
#include<iostream>
using namespace std;

int main(){
    
    int x, y;
    while(cin >> x >> y){
        while(x!=y){
            if(x < y) y/=2;
            else x/=2;
        }
        cout << x << endl;
    }
}
```

##### 单词替换 KY46

```C++
#include<iostream>
#include<string>
#include<vector>

using namespace std;
vector<string> v;

int main(){
    string s;
    while(cin >> s){
        v.push_back(s);
        if(getchar() == '\n') break;  //匹配缓冲区中的回车
    }
    string s1, s2;
    cin >> s1 >> s2;
    for(int i = 0; i < v.size(); ++i){
        if(v[i] == s1) cout << s2 << " ";
        else cout << v[i] << " ";
    }
    cout << endl;
}
```

##### 密码翻译 KY33

```C++
#include<iostream>
#include<string>
using namespace std;

string code(string s){
    for(int i = 0; i < s.size(); ++i){
        if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')){
            s[i]++;
            //溢出处理  小写字母的ASCII码在大写字母的后面
            if(s[i] > 'Z' && s[i] < 'a' || s[i] > 'z'){
                s[i] -= 26;
            }
        }
    }
    return s;
}

int main(){
    string s;
    getline(cin, s);
    string ans = code(s);
    cout << ans << endl;
}
```
