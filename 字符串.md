```C++
#include<iostream>
#include<string>
using namespace std;

string code(string s){
    for(int i = 0; i < s.size(); ++i){
        if((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')){
            s[i]++;
            //溢出处理  小写字母的ASCII码在大写字母的后面
            if(s[i] > 'Z' && s[i] < 'a' || s[i] > 'z'){
                s[i] -= 26;
            }
        }
    }
    return s;
}

int main(){
    string s;
    getline(cin, s);
    string ans = code(s);
    cout << ans << endl;
}
```

##### 首字母大写 KY57

```C++
// 先处理首个单词  再遍历剩余的字符串
#include<iostream>
#include<string>

using namespace std;

int main(){
    string s;
    
    while(getline(cin, s)){
        //处理首字母
        if(s[0] >= 'a' && s[0] <= 'z') s[0] = s[0] - 32;
        
        for(int i = 1; i < s.size(); ++i){
            if((s[i-1] == ' ' || s[i-1] == '\t' ||
              s[i-1] == '\r' || s[i-1] == '\n') && (s[i] >= 'a' && s[i] <= 'z')){
                s[i] -= 32;
            }
              
        }
        cout << s << endl;
    }
} 
```

##### 子串计算 KY68

```C++
#include<iostream>
#include<map>
#include<string>
using namespace std;

int main(){
    map<string, int> m;
    string str;
    cin >> str;
    for(int i = 0; i < str.size(); ++i){
        for(int j = i; j < str.size(); ++j){
            m[str.substr(i,j-i+1)]++;
        }
    }
    for(map<string, int>::iterator it = m.begin(); it != m.end(); ++it){
        if(it->second > 1) cout << it->first << " " << it->second << endl;
    }
    return 0;
}
```

##### 特殊乘法 KY18

```C++
#include<iostream>
#include<string>
using namespace std;

int main(){
    string s1, s2;
    while(cin >> s1 >> s2){
        int sum = 0;
        for(int i = 0; i < s1.size(); ++i){
            for(int j = 0; j < s2.size(); ++j){
```

##### 浮点数加法 KY79

```C++
#include<iostream>
#include<string>
using namespace std;

int main(){
    string s1, s2;
    while(getline(cin, s1)){
        getline(cin, s2);
        //将两字符串补到相同长度
        int pos1 = s1.find('.');
        int pos2 = s2.find('.');
        if(pos1 > pos2) s2.insert(0, pos1-pos2, '0');
        else s1.insert(0, pos2-pos1, '0');
        
        if(s1.size() > s2.size()) s2.insert(s2.size(), s1.size()-s2.size(), '0');
        else s1.insert(s1.size(), s2.size()-s1.size(), '0');
        
        int carry = 0;
        for(int i = s1.size()-1; i >= 0; --i){
            if(s1[i] == '.') continue;
            int temp = s1[i] - '0' + s2[i] - '0' + carry;
            carry = temp/10;
            s1[i] = temp%10 + '0';
        }
        if(carry != 0) s1.insert(0, 1, '1');
        cout << s1 << endl;
    }
}
```

##### skew数 KY51

```C++
#include<iostream>
#include<string>
#include<cmath>
using namespace std;

int main(){
    string s;
    while(cin >> s){
        if(s == "0") break;
        int n = s.size();
        long long sum = 0;
        //这里不能用n 只能用s.size()
        for(int i = 0; i < s.size(); ++i){
            //记得n--
            sum += (s[i] - '0') *(pow(2, n--)-1);
        }
        cout << sum << endl;
    }
}
```
