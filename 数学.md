##### n的阶乘 KY17

```C++
#include<iostream>
using namespace std;

int main(){
    int n;
    while(cin >> n){
        long long ans = n;  //int范围不足
        while(--n){
            ans *= n;
        }
        cout << ans << endl;
    }
}
```

##### 最简真分数 KY35

```C++
#include<iostream>
using namespace std;
//最大公约数
int gcd(int a, int b){
    if(b == 0) return a;
    else return gcd(b, a%b);
}

int arr[605];

int main(){
    int n;
    while(cin >> n){
        if(n == 0) break;
        for(int i = 0; i < n; ++i){
            cin >> arr[i];
        }
        int sum = 0;
        for(int i = 0; i < n; ++i){
            for(int j = i+1; j < n; ++j){ //不用再去匹配前面的
                if(gcd(arr[i], arr[j]) == 1) sum++; //不要写成i==j
            }
        }
        cout << sum << endl;
    }
}
```
