##### 最简真分数 KY35

```C++
#include<iostream>
using namespace std;
//最大公约数
int gcd(int a, int b){
    if(b == 0) return a;
    else return gcd(b, a%b);
}

int arr[605];

int main(){
    int n;
    while(cin >> n){
        if(n == 0) break;
        for(int i = 0; i < n; ++i){
            cin >> arr[i];
        }
        int sum = 0;
        for(int i = 0; i < n; ++i){
            for(int j = i+1; j < n; ++j){ //不用再去匹配前面的
                if(gcd(arr[i], arr[j]) == 1) sum++; //不要写成i==j
            }
        }
        cout << sum << endl;
    }
}
```

##### 约数的个数 KY3

```C++
#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
#include<cstring>

using namespace std;

int main(){
    int n;
    while(cin >> n){
        for(int i = 0; i < n; ++i){
            int x;
            cin >> x;
            int answer = 0;
            int bound = sqrt(x);
            for(int j = 1; j <= bound; ++j ){
                if(x % j == 0) answer+=2;  //若存在一个小于sqrt(x)的约数 则对应一个大于sqrt(x)的约数
            }
            
            if(bound*bound == x ) answer--; //多加了一个
            cout << answer << endl;
        }
        
    }    
}
```

##### 质因数的个数 KY7

```C++
#include<iostream>
#include<vector>
#include<cmath>
#include<cstring>
using namespace std;

const int maxn = sqrt(1e9) + 1;  //只需筛选到sqrt+1
vector<int> prime;
int isprime[31623];

void Initial(){
    memset(isprime, 1, sizeof(isprime));
    //记得处理1和0
    isprime[0] = 0;
    isprime[1] = 0;
    for(int i = 2; i < maxn; ++i){
        if(!isprime[i]) continue;  //加速
        prime.push_back(i);
        //从i*i开始
        for(int j = i * i; j < maxn; j+=i){
            isprime[j] = 0;
        }
    }
}

int main(){
    Initial();
    int n;
    cin >> n;
    int answer = 0;
    for(int i = 0; prime[i] < n && i < prime.size(); ++i){
        //这里不要写成n%i
        while(n % prime[i] == 0){
            n /= prime[i];
            answer++;
        }
    }
    if(n > 1) answer++;
    cout << answer << endl;
}
```

##### 打印极值点下标

```C++
#include<iostream>
using namespace std;

int main(){
    int n, i, left, mid, right;
    
    while(cin >> n){
        cin >> mid >> right;
        if(mid != right) cout << "0 ";
        // 这里的i需要在外面声明
        for(i = 1; i < n-1; ++i){
            left = mid;
            mid = right;
            cin >> right;
            //判断条件 比两边大或比两边小
            if((mid - left) * (mid - right) > 0) cout << i << " ";
        }
        
        if(mid != right) cout << i;
        cout << endl;
    }
}
```

##### 进制转换 KY30

```C++

#include<iostream>
#include<string>
#include<vector>
using namespace std;

string Divide(string s, int x){
    int carry = 0;
    for(int i = 0; i < s.size(); ++i){
        int current = carry*10 + (s[i]-'0');
        s[i] = current/x + '0';
        carry = current%x;  //这里是%x 不是%10
    }
    int pos = 0;
    while(s[pos++]=='0'); //去除前面的0
    return s.substr(pos-1);

}

int main(){
    string s;

    while(cin >> s){
        vector<int> v;
        while(s.size() != 0){ //因为最后把零删掉了 所以最后会返回空串
            v.push_back((s[s.size()-1]-'0')%2);
            s = Divide(s, 2);
        }
        
        for(int i = v.size()-1; i >= 0; --i) cout << v[i];
        cout << endl;
    }
}
```
