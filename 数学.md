##### n的阶乘 KY17

```C++
#include<iostream>
using namespace std;

int main(){
    int n;
    while(cin >> n){
        long long ans = n;  //int范围不足
        while(--n){
            ans *= n;
        }
        cout << ans << endl;
    }
}
```

##### 最简真分数 KY35

```C++
#include<iostream>
using namespace std;
//最大公约数
int gcd(int a, int b){
    if(b == 0) return a;
    else return gcd(b, a%b);
}

int arr[605];

int main(){
    int n;
    while(cin >> n){
        if(n == 0) break;
        for(int i = 0; i < n; ++i){
            cin >> arr[i];
        }
        int sum = 0;
        for(int i = 0; i < n; ++i){
            for(int j = i+1; j < n; ++j){ //不用再去匹配前面的
                if(gcd(arr[i], arr[j]) == 1) sum++; //不要写成i==j
            }
        }
        cout << sum << endl;
    }
}
```

##### 约数的个数 KY3

```C++
#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
#include<cstring>

using namespace std;

int main(){
    int n;
    while(cin >> n){
        for(int i = 0; i < n; ++i){
            int x;
            cin >> x;
            int answer = 0;
            int bound = sqrt(x);
            for(int j = 1; j <= bound; ++j ){
                if(x % j == 0) answer+=2;  //若存在一个小于sqrt(x)的约数 则对应一个大于sqrt(x)的约数
            }
            
            if(bound*bound == x ) answer--; //多加了一个
            cout << answer << endl;
        }
        
    }    
}
```

##### 质因数的个数 KY7

```C++
#include<iostream>
#include<vector>
#include<cmath>
#include<cstring>
using namespace std;

const int maxn = sqrt(1e9) + 1;  //只需筛选到sqrt+1
vector<int> prime;
int isprime[31623];

void Initial(){
    memset(isprime, 1, sizeof(isprime));
    //记得处理1和0
    isprime[0] = 0;
    isprime[1] = 0;
    for(int i = 2; i < maxn; ++i){
        if(!isprime[i]) continue;  //加速
        prime.push_back(i);
        //从i*i开始
        for(int j = i * i; j < maxn; j+=i){
            isprime[j] = 0;
        }
    }
}

int main(){
    Initial();
    int n;
    cin >> n;
    int answer = 0;
    for(int i = 0; prime[i] < n && i < prime.size(); ++i){
        //这里不要写成n%i
        while(n % prime[i] == 0){
            n /= prime[i];
            answer++;
        }
    }
    if(n > 1) answer++;
    cout << answer << endl;
}
```
