##### Zero-complexity Transposition KY109 栈

```C++
//可以直接用数组
#include<iostream>
#include<string>
#include<stack>
using namespace std;

string s[10005];
stack<string> st;

int main(){
    int n;
    while(cin >> n){
        string s;
        for(int i = 0; i < n; ++i){
            cin >> s;
            st.push(s);
        }
        while(!st.empty()){
            cout << st.top() << " ";
            st.pop();
        }
        cout << endl;
    }
}
```
