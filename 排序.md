##### 求最大最小数 KY13

```C++
#include<iostream>
using namespace std;

int main(){
    int n;
    while(cin >> n){
        int max = -1e6, min = 1e6;  //赋值别赋错 max赋最小值
        for(int i = 0; i < n; ++i){
            int x;
            cin >> x;
            if(x > max) max = x;
            if(x < min) min = x;
        }
        cout << max << " " << min << endl;
    }
}
```

##### 成绩排序 KY9

```C++
#include<iostream>
#include<algorithm>
using namespace std;

struct Student{
    int no;
    int grade;
}stu[105];

bool cmp(Student x, Student y){
    if(x.grade != y.grade) return x.grade < y.grade;
    else return x.no < y.no;
}

int main(){
    int n;
    while(cin >> n){
        for(int i = 0; i < n; ++i) cin >> stu[i].no >> stu[i].grade;
        sort(stu, stu+n, cmp);
        for(int i = 0; i < n; ++i){
            cout << stu[i].no << " " << stu[i].grade << endl;
        }
    }
}
```

##### 成绩排序 KY2  使用稳定排序

```C++
#include<algorithm>
#include<string>

    string name;
    int grade;
}stu[1005];

bool compareDesc(Student a, Student b){
    return a.grade > b.grade;
}

bool compareAsc(Student a, Student b){
    return a.grade < b.grade;
}

int main(){
    int n, order;
    while(cin >> n){
        cin >> order;
        for(int i = 0; i < n; ++i){
            cin >> stu[i].name >> stu[i].grade;
            }

        if(order == 0) stable_sort(stu, stu+n, compareDesc); //稳定排序
        else stable_sort(stu, stu+n, compareAsc);

        for(int i = 0; i < n; ++i){
            cout << stu[i].name << " " << stu[i].grade << endl;
        }
    }
}
```

##### 成绩排序 KY215 三级排序

```C++
#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

struct Student{
    string name;
    int age;
    int grade;
}stu[1005];

bool cmp(Student a, Student b){
    if(a.grade != b.grade) return a.grade < b.grade;
    else if(a.name != b.name) return a.name < b.name;
    else return a.age < b.age;
}

int main(){
    int n;
    while(cin >> n){
        for(int i = 0; i < n; ++i){
            cin >> stu[i].name >> stu[i].age >> stu[i].grade;
        }

        sort(stu, stu+n, cmp); 
        
        for(int i = 0; i < n; ++i){
            cout << stu[i].name << " " << stu[i].age << " " << stu[i].grade << endl;
        }
    }
}
```

##### 整数奇偶排序 KY67

```C++
#include<iostream>
#include<algorithm>
using namespace std;

//思路   如果奇偶性不同 奇数在前(cmp返回1顺序不变) 奇偶相同时，偶数从小到大 奇数从大到小
bool cmp(int x, int y){
    if(x%2 != y%2) return x%2;
    else if(x%2 == 0) return x < y;
    else return x > y;
}

int arr[10];

int main(){
    for(int i = 0; i < 10; ++i){
        cin >> arr[i];
    }
    
    sort(arr, arr+10, cmp);
    
    for(int i = 0; i < 10; ++i){
        cout << arr[i] << " ";
    }
    cout << endl;
}
```

##### 特殊排序 KY211

```C++
#include<iostream>
#include<algorithm>
using namespace std;

int arr[1005];

int main(){
    int n;
    while(cin >> n){
        for(int i = 0; i < n; ++i){
            cin >> arr[i];
        }
        sort(arr, arr+n);
        cout << arr[n-1] << endl;
        if(n == 1) cout << -1 << endl;
        else{
            for(int i = 0; i < n-1; ++i) cout << arr[i] << " ";
        }
        cout << endl;
    }
}
```
