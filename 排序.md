##### 求最大最小数 KY13

```C++
#include<iostream>
using namespace std;

int main(){
    int n;
    while(cin >> n){
        int max = -1e6, min = 1e6;  //赋值别赋错 max赋最小值
        for(int i = 0; i < n; ++i){
            int x;
            cin >> x;
            if(x > max) max = x;
            if(x < min) min = x;
        }
        cout << max << " " << min << endl;
    }
}
```
