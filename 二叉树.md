##### 二叉树遍历 KY11

```C++
//字符串建树
#include<iostream>
#include<string>
using namespace std;

struct TreeNode{
    char data;
    TreeNode* lchild;
    TreeNode* rchild;
    TreeNode(char c): data(c), lchild(NULL), rchild(NULL) {}
};

TreeNode* Build(string s, int& pos){  //pos为整型引用
    char c = s[pos++];
    if(c == '#') return NULL;
    TreeNode* root = new TreeNode(c);
    root->lchild = Build(s, pos);
    root->rchild = Build(s, pos);
    return root;  //记得返回
}

void InOrder(TreeNode* root){  //root为指针
    if(root == NULL) return;
    InOrder(root->lchild);
    cout << root->data << " ";
    InOrder(root->rchild);
}

int main(){
    string s;
    while(cin >> s){
        int pos = 0;
        TreeNode* t = Build(s, pos);
        InOrder(t);
        cout << endl;
    }
}
```

###### 层次遍历

```C++
#include<iostream>
#include<string>
#include<queue>
using namespace std;

struct TreeNode{
    char data;
    TreeNode* lchild;
    TreeNode* rchild;
    TreeNode(char c): data(c), lchild(NULL), rchild(NULL) {}
};

TreeNode* Build(string s, int& pos){
    char c = s[pos++];
    if(c == '#') return NULL;
    TreeNode* root = new TreeNode(c);
    root->lchild = Build(s, pos);
    root->rchild = Build(s, pos);
    return root;
}

int main(){
    string s;
    while(cin >> s){
        int pos = 0;
        TreeNode* t = Build(s, pos);
        queue<TreeNode*> q;
        if(t != NULL) q.push(t);
        while(!q.empty()){
            TreeNode* now = q.front();
            q.pop();
            cout << now->data << " ";
            if(now->lchild != NULL) q.push(now->lchild);
            if(now->rchild != NULL) q.push(now->rchild);
        }
    }
}
```

##### 二叉树遍历

```C++
//根据前序和中序序列建树
#include<iostream>
#include<string>
#include<queue>
using namespace std;

struct TreeNode{
    char data;
    TreeNode* lchild;
    TreeNode* rchild;
    TreeNode(char c): data(c), lchild(NULL), rchild(NULL) {}
};

TreeNode* Build(string s1, string s2){
    if(s1.size() == 0) return NULL;
    char c = s1[0];
    TreeNode* root = new TreeNode(c);
    int pos = s2.find(c);
    //substr(a,b) 表示从下标a开始截取b个字符 
    root->lchild = Build(s1.substr(1, pos), s2.substr(0,pos));
    root->rchild = Build(s1.substr(pos+1), s2.substr(pos+1));
    return root;
}

void PostOrder(TreeNode* root){
    if(root == NULL) return;
    PostOrder(root->lchild);
    PostOrder(root->rchild);
    cout << root->data;
}

int main(){
    string s1, s2;
    while(cin >> s1 >> s2){
        TreeNode* t = Build(s1, s2);
        PostOrder(t);
        cout << endl;
    }
}
```

##### 二叉搜索树 KY124

```C++
#include<iostream>
using namespace std;

struct TreeNode{
    int data;
    TreeNode* lchild;
    TreeNode* rchild;
    TreeNode(int x): data(x), lchild(NULL), rchild(NULL) {}
};

TreeNode* insert(TreeNode* root, int x){
    if(root == NULL){
        root = new TreeNode(x);
    }else if(x < root->data){
        root->lchild = insert(root->lchild, x);
    }else{
        root->rchild = insert(root->rchild, x);
    }
    return root;
}

bool isSame(TreeNode* t1, TreeNode* t2){
    if(t1 == NULL && t2 == NULL) return true;
    if((t1 == NULL && t2 != NULL) || (t1 != NULL && t2 == NULL)) return false;
    if(t1->data != t2->data) return false;
    bool l = isSame(t1->lchild, t2->lchild);
    bool r = isSame(t1->rchild, t2->rchild);
    return l && r;
}

int main(){
    int n;
    string s;
    while(cin >> n){
        cin >> s;
        TreeNode* root = NULL;
        for(int i = 0; i < s.size(); ++i) root = insert(root, s[i]);
        for(int i = 0; i < n; ++i){
            TreeNode* root1 = NULL;
            string s1;
            cin >> s1;
            for(int i = 0; i < s1.size(); ++i) root1 = insert(root1, s1[i]);
            if(isSame(root, root1)){
                cout << "YES" << endl;
            }else{
                cout << "NO" << endl;
            }
        }
    }
}
```
